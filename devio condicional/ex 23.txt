var
   a,b,c: inteiro
   validacao,eq,es,is: logico
inicio
      escreva(" digite o lado A ")
      leia(a)
      escreva(" digite o lado B ")
      leia(b)
      escreva(" digite o lado C ")
      leia(c)
      validacao := (a+b>c) e (b+c>a) e (c+a>b)
      eq := (a=b) e (b=c)
      es := (a<>b) e (b<>c) e (c<>a)
      is := (a=b) ou (b=c) ou (c=a)
      escreval(" os 3 lados acima configuram um triângulo : ",validacao)
      escreval(" o triângulo é equilátero : ",eq)
      escreval(" o triângulo é escalêno : ",es)
      escreval(" o triângulo é isoseles : ",is)
