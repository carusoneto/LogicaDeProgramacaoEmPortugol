   c, j, par : inteiro
inicio

      escreval(" digite os valores da matriz : ")
      para c := 1 ate 3 passo 1 faca
           para j := 1 ate 3 passo 1 faca

                   par := par + 1
                fimse
           fimpara
      fimpara

      para c := 1 ate 3 passo 1 faca
           para j := 1 ate 3 passo 1 faca
                 se ( v[c,j] % 2 = 0 ) entao
                      escreva("{",v[c,j],"}")
                 senao
                      escreva(v[c,j])
