algoritmo "semnome"
var
   n : vetor[1..10] de inteiro
   c, i, aux : inteiro
inicio
      escreval(" digite 10 números aleatorios : ")
      para c := 1 ate 10 passo 1 faca
           leia(n[c])
      fimpara
      escreval(" o intervalo de números acima, digitados de forma ordenada progressiva é : ")
      para c := 1 ate 9 passo 1 faca
           para i := c + 1 ate 10 passo 1 faca
                se (n[c] > n[i] ) entao
                   aux := n[c]
                   n[c] := n[i]
                   n[i] := aux
                fimse
           fimpara
      fimpara
      escreval(" ")
      para c := 1 ate 10 passo 1 faca
           escreva(n[c])
      fimpara
      escreval(" ")
      escreval(" e a ordenação regressiva é : ")
      para c := 10 ate 1 passo -1 faca
           escreva(n[c])
      fimpara
fimalgoritmo
