algoritmo "semnome"
var
   n : caracter
   soc : vetor [1..10] de caracter
   c, tot : inteiro
inicio
      escreval(" digite 10 nomes : ")
                 para c := 1 ate 10 passo 1 faca
                      leia(n)
                      se (copia(n,1,1) = "c")   entao
                         tot := tot + 1
                         soc[tot] := n
                      fimse
                 fimpara
                 escreval(" me escreva somente os nomes que começãm com a letra c ")
                 para c := 1 ate tot passo 1 faca
                      escreval(soc[c])
                 fimpara
fimalgoritmo
