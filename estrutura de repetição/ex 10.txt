algoritmo "semnome"
var
   contagem, n, soma, fim, maior: inteiro
inicio
      escreva(" quantas vezes você quer somar : ")
      leia(contagem)
      soma := 0
      maior := 0
      enquanto (contagem >= 0) faca
               escreva(" digite o número a ser somado : ")
               leia(n)
               se ( n > maior ) entao
                  maior := n
               fimse
               soma := soma + n
               contagem := contagem - 1
      fimenquanto
      escreval(" a soma entre todos os números digitados é ",soma)
      escreva(" digite o maior número escrito : ",maior)
fimalgoritmo
